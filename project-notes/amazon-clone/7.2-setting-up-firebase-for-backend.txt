- goto firebase.js in your local project folder.... add these codes to....initialize firebase.... setup database and handle sign ins.... then export these....

    const firebaseApp = firebase.initializeApp(firebaseConfig);

    const db = firebaseApp.firestore();

    const auth = firebase.auth()

    export { db, auth };
